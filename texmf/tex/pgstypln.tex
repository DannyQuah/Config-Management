%% For pagestyle fancy, LaTeX Companion 4.3.2 pp. 96ff
%

\lhead[\fancyplain{}{}]{\fancyplain{}{}}%
\rhead[\fancyplain{}{}]{\fancyplain{}{}}%
\cfoot[\fancyplain{}{\rm--\thepage--}]{\fancyplain{}{\rm--\thepage--}}%
\chead[\fancyplain{}{}]{\fancyplain{}{}}%
\lfoot[\fancyplain{}{}]%
 {\fancyplain{}{}}%
 \rfoot[\fancyplain{}{}]{\fancyplain{}{}}
