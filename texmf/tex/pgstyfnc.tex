 \pagestyle{fancyplain}%
%
%% For pagestyle fancy, LaTeX Companion 4.3.2 pp. 96ff
%
%% \setlength{\headrulewidth}{0pt}% 0.4pt default
%% \setlength{\footrulewidth}{0pt}% 0pt default
 \renewcommand{\headrulewidth}{0pt}% 0.4pt default
 \renewcommand{\footrulewidth}{0pt}% 0pt default
%% %% begin boolean{qPgHead}
 \ifthenelse{\boolean{qPgHead}}{%
%  \setlength{\headrulewidth}{0.2pt}% 0.4pt default
  \renewcommand{\headrulewidth}{0.2pt}% 0.4pt default
  \ifthenelse{\boolean{qTwoSide}}{%
   \lhead[\fancyplain{}{\qPageHead \hfill \textbf{\thepage}}]{\fancyplain{}{}}%
   \rhead[\fancyplain{}{}]{\fancyplain{}{\textbf{\thepage} \hfill \qPageHead}}%
   \cfoot[\relax]{\relax}%
   \chead{}\lfoot{}\rfoot{}% Set all these to nothing
  }{%
   \lhead[\fancyplain{}{}]{\fancyplain{}{}}%
   \rhead[\fancyplain{}{\qPageHead}]{\fancyplain{}{\qPageHead}}%
   \cfoot[\fancyplain{}{\textrm{--\thepage--}}]%
         {\fancyplain{}{\textrm{--\thepage--}}}%
   \chead{}\lfoot{}\rfoot{}%
  }%\boolean{qTwoSide}
%%
 }{%!\boolean{qPgHead}
  \lhead[\fancyplain{}{}]{\fancyplain{}{}}%
  \rhead[\fancyplain{}{}]{\fancyplain{}{}}%
  \cfoot[\fancyplain{}{\textrm{--\thepage--}}]%
        {\fancyplain{}{\textrm{--\thepage--}}}%
  \chead{}\lfoot{}\rfoot{}%
 }%\boolean{qPgHead}
%% %% finish boolean{qPgHead}
% \addtolength{\headheight}{2pt}%
 \addtolength{\headheight}{3pt}%
% Restore
 \thispagestyle{plain}
